\section{Normalized System of ODEs}
Al hacer las siguientes transformaciones sobre las ecuaciones \eqref{eq:ODE1}  y \eqref{eq:ODE2}

\begin{eqnarray*}
	\rho = \dfrac{r}{R}, \quad \quad \hat{n}(\rho) = \dfrac{n(r)}{n_{max}} , \quad \quad \hat{T}(\rho) = \dfrac{T(r)}{T_{max}}	
\end{eqnarray*}

se obtiene un sistema de ecuaciones diferenciales cuyas escalas son similares y ecuaciones adimensionales, esto será de gran ayuda para la PINN.

\begin{eqnarray}
	\frac{d}{d\rho}\left[\rho \hat{D}_\perp \frac{d\hat{n}}{d\rho}\right] &+& \frac{R^2n_{max}}{D_0}[ \langle \sigma v \rangle_{rec}(T_{max})\hat{R}_{rec}(\hat{T}) + \langle \sigma v \rangle_{ion_2}(T_{max})\hat{R}_{ion_2}(\hat{T}) + \langle \sigma v \rangle_{rad}^i(T_{max})\hat{R}^i_{rad}(\hat{T})] \rho\hat{n}^2 \nonumber \\
	&+& \frac{R^2 n_0}{D_0}[ \langle \sigma v \rangle_{ion}(T_{max})\hat{R}_{ion}(\hat{T}) + \langle \sigma v \rangle^0_{rad}(T_{max})\hat{R}^0_{rad}(\hat{T})]\rho \hat{n} = 0 \quad \blacksquare
\end{eqnarray}

con 

\begin{eqnarray}
	D_0 &=& \frac{2 \sqrt{2 \pi m}}{3}\frac{e^{3/2}}{(4\pi \varepsilon_0 B)^2}\frac{n_{max}}{\sqrt{T_{max}}} \\
	b &=&  \log{\left[\frac{(4\pi)^2}{n_{max}}\left(\frac{\varepsilon_0 T_{max}}{e}\right)^3\right]} \\
	\hat{D}_\perp(\hat{n},\hat{T}) &=& \frac{\hat{n}}{\sqrt{\hat{T}}} \left[ b + \log{\left(\frac{\hat{T^3}}{\hat{n}}\right)} \right]\\
	R_{proc}(\hat{T}) &=& \frac{\langle \sigma v\rangle_{ion}(T_{max}\hat{T})}{\langle \sigma v\rangle_{ion}(T_{max})}
\end{eqnarray}

\begin{eqnarray}
	&& \frac{d}{d\rho}\left(\rho \hat{n} \hat{D}_\perp\frac{d\hat{T}}{d\rho} \right) + \frac{15}{47}\rho\left(\frac{d\hat{n}}{d\rho}\right)\left(\frac{d\hat{T}}{d\rho}\right) + \frac{15}{47}\hat{T}\frac{d}{d\rho}\left(\rho \hat{D}_\perp \frac{d\hat{n}}{d\rho}\right) \nonumber \\ 
	&+& \frac{10}{47}\frac{P_{ext}}{\sigma\sqrt{2\pi}}\frac{eR^2}{n_{max}T_{max}D_0}\rho\exp{\left[-\frac{1}{2}\left(\frac{R}{\sigma}\right)^2(\rho - \langle \rho \rangle)^2\right]} \nonumber \\
	&-& \frac{10}{47}\frac{e n_{max}R^2}{T_{max} D_0}\left[ E_{ion_2}\langle \sigma v \rangle_{ion_2}(T_{max})\hat{R}_{ion_2}(\hat{T}) + E_{rec}\langle \sigma v \rangle_{rec}(T_{max})\hat{R}_{rec}(\hat{T}) + E_{rad}^i\langle \sigma v \rangle_{rad}^i(T_{max})\hat{R}_{rad}^i(\hat{T})\right]\rho\hat{n}^2 \nonumber \\
	&-& \frac{10}{47}\frac{e n_0R^2}{T_{max}D_0}\left[ E_{ion}\langle \sigma v \rangle_{ion}(T_{max})\hat{R}_{ion}(\hat{T}) + E_{rad}^0\langle \sigma v \rangle_{ion}(T_{max})\hat{R}_{rad}^0(\hat{T}) \right]\rho\hat{n} \nonumber \\ 
	&-&\frac{15}{47}\frac{e^3(ZBR)^2}{mT_{max}}\frac{\rho \hat{n}^2}{\sqrt{\hat{T}}}\left[ b + \log{ \left( \frac{\hat{T}^3}{\hat{n}}\right)} \right] \quad \blacksquare
\end{eqnarray}

donde,

\begin{eqnarray}
	\kappa_0 &=& \frac{47\sqrt{2\pi m}}{15}\frac{e^{3/2}}{(4\pi \varepsilon_0 B)^2}\frac{\rho n^2_{max}}{\sqrt{T_{max}}} = \frac{47}{10}n_{max}D_0\\
	\hat{\kappa}(\hat{n}, \hat{T}) &=& \frac{\hat{n}^2}{\sqrt{\hat{T}}} \left[ b + \log{\left(\frac{\hat{T}^3}{\hat{n}} \right)}\right] = \hat{n}\hat{D}_\perp(\hat{n}, \hat{T})
\end{eqnarray}

Es importante considerar que la temperatura $T_{max}$ tiene unidades de $eV$, la densidad $n_{max}$ de $cm^{-3}$ y finalmente el radio $R$ unidades de $cm$.
